
// Como comparar listas 
// .every()

const array = [4, 6, 7, 8, 3, 6, 2, 1, -4, -7, 12, 5, -40]

const resultado = array.every(valor => {
    if (typeof valor === "number") {
        return true
    } else {
        return false
    }
})
console.log(resultado)
